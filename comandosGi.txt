git init --> Informa ao git que a pasta deve ser controla, será usado no controle de versão
	git ls-files --> Comando que retorna os arquivos controlados pelo Git.
	git add <file-name> --> Adiciona um arquivo específico que não está rastreado ao controle de versão.
	git add --all -- > Adiciona todos os arquivo que não estão rastreados ao controle de versão.
	git commit -m <Mensagem> --> Realiza a gravação no repositório.
